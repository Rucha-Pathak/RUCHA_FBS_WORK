import java.util.*;

class QueueUsingLinkeList
{
	LinkedList<Integer> l2 = new LinkedList<Integer>();
	
	 void push(int d)//to add element into stack
	 {
		 l2.addLast(d);
		 System.out.println(d + " added susccessfully");
	 }
	
	 int pop()//to remove element from stack
	 {
		 if(l2.isEmpty())
		 {
			 System.out.println("Queue is Empty");
			 return -1;
		 }
		 int r = l2.removeFirst();
		 System.out.println(r+ " removed from Queue !");
		 return r;
	 }
	 
	 int peek()//to display top element of stack
	 {
		 if(l2.isEmpty())
		 {
			 System.out.println("Queue is empty");
			 return -1;
		 }
		  return l2.getFirst();
	 }
	 
	 boolean isEmpty()//to check wheather stack is empty or not
	 {
		  return l2.isEmpty();
	 }
	 void display()
	 {
		 System.out.println("Queue elements are: "+l2);
	 }
}
public class TestQueue {

	public static void main(String[] args) 
	{
		// TODO Auto-generated method stub
		QueueUsingLinkeList Queue = new QueueUsingLinkeList();
		Queue.push(10);
		Queue.push(20);
		Queue.push(30);
		Queue.push(40);
		Queue.display();
		Queue.pop();
		Queue.display();
		System.out.println("First element of Queue : "+Queue.peek());
		Queue.push(50);
		Queue.push(60);
		Queue.push(70);
		Queue.display();
		Queue.pop();
		System.out.println("First element of Queue "+Queue.peek());
		
		Queue.display();

	}

}
